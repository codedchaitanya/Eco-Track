# eco_dashboard.py

# --- Libraries ---
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
import plotly.express as px
import streamlit as st

# --- Load Dataset ---
@st.cache_data
def load_data():
    df = pd.read_csv("company_esg_financial_dataset.csv")
    return df

df = load_data()

# --- Cleaning ---
# Drop duplicates
df = df.drop_duplicates()

# Handle missing values (example: fill numeric with mean, categorical with mode)
for col in df.select_dtypes(include=np.number).columns:
    df[col].fillna(df[col].mean(), inplace=True)
for col in df.select_dtypes(exclude=np.number).columns:
    df[col].fillna(df[col].mode()[0], inplace=True)

# --- EDA ---
st.title("🌍 Eco-Track: Company Sustainability Dashboard")

st.subheader("Raw Dataset Preview")
st.write(df.head())

st.subheader("Dataset Info")
st.write(df.describe())

# Correlation Heatmap
st.subheader("Correlation Heatmap (Numerical Features)")
corr = df.corr(numeric_only=True)
fig_corr = px.imshow(corr, text_auto=True, aspect="auto", color_continuous_scale="RdBu_r")
st.plotly_chart(fig_corr)

# Company-wise sustainability score trend
if "Year" in df.columns and "Sustainability_Score" in df.columns and "Company" in df.columns:
    st.subheader("📈 Company Progress in Sustainability Over Years")
    company = st.selectbox("Select Company", df["Company"].unique())
    df_company = df[df["Company"] == company]
    fig_line = px.line(df_company, x="Year", y="Sustainability_Score", title=f"{company} Sustainability Trend", markers=True)
    st.plotly_chart(fig_line)

# Resource Utilization Comparison
if "Energy_Usage" in df.columns and "Water_Usage" in df.columns:
    st.subheader("💧 Resource Utilization Comparison")
    fig_scatter = px.scatter(df, x="Energy_Usage", y="Water_Usage", color="Company",
                             size="Sustainability_Score", hover_name="Company")
    st.plotly_chart(fig_scatter)

# Top Sustainable Companies
if "Sustainability_Score" in df.columns:
    st.subheader("🏆 Top Sustainable Companies")
    top_companies = df.groupby("Company")["Sustainability_Score"].mean().sort_values(ascending=False).head(10)
    fig_bar = px.bar(top_companies, x=top_companies.index, y=top_companies.values, title="Top Companies by Sustainability")
    st.plotly_chart(fig_bar)
